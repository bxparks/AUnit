46  // The LifeCycle states are used by TestRunner to determine what a Test
47  // should do. Unlike the assertion Status, the LfeCycle is mostly hidden
48  // from client code. The state transition diagram looks like this:
49  //
50  // include()/exclude()
51  // .---------------------> Excluded -----------.
52  // v v
53  // New Finished -> (out of list)
54  // \ setup() assertion() teardown() ^
55  // -------> Setup -------> Asserted ----------'
56 
57  enum class LifeCycle : uint8_t
58  {
60  New = 0,
61 
68  Excluded = 1,
69 
77  Setup = 2,
78 
83  Asserted = 3,
84 
91  Finished = 4,
92  };

94  // The assertion Status is the result of an "assertion()". In addition to
95  // the usual pass() and fail(), there are meta-assertions such as skip()
96  // and expire(). When the Status is changed from Unknown, the
97  // lifeCycle state changes to Asserted.
