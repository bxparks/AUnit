181  void runTest() {
182  setupRunner();
183 
184  // Print initial header if this is the first run.
185  if (!mIsRunning) {
186  printStartRunner();
187  mIsRunning = true;
188  }
189 
190  // If no more test cases, then print out summary of run.
191  if (*Test::getRoot() == nullptr) {
192  if (!mIsResolved) {
193  mEndTime = std::chrono::high_resolution_clock::now();
194  resolveRun();
195  mIsResolved = true;
196  #if EPOXY_DUINO
197  exit((mFailedCount || mExpiredCount) ? 1 : 0);
198  #endif
199  }
200  return;
201  }
202 
203  // If reached the end and there are still test cases left, start from the
204  // beginning again.
205  if (*mCurrent == nullptr) {
206  mCurrent = Test::getRoot();
207  }
208 
209  // Implement a finite state machine that calls the (*mCurrent)->setup() or
210  // (*mCurrent)->loop(), then changes the test case's mStatus.
211  switch ((*mCurrent)->getLifeCycle()) {
212  case Test::LifeCycle::New:
213  // Transfer the verbosity of the TestRunner to the Test.
214  (*mCurrent)->enableVerbosity(mVerbosity);
215  (*mCurrent)->setup();
216 
217  // Support assertXxx() statements inside the setup() method by
218  // moving to the next lifeCycle state if an assertXxx() did not fail
219  // inside the setup().
220  if ((*mCurrent)->getLifeCycle() == Test::LifeCycle::New) {
221  (*mCurrent)->setLifeCycle(Test::LifeCycle::Setup);
222  }
223  break;
224  case Test::LifeCycle::Excluded:
225  // If a test is excluded, go directly to LifeCycleFinished, without
226  // calling setup() or teardown().
227  (*mCurrent)->enableVerbosity(mVerbosity);
228  (*mCurrent)->setStatus(Test::Status::Skipped);
229  mSkippedCount++;
230  (*mCurrent)->setLifeCycle(Test::LifeCycle::Finished);
231  break;
232  case Test::LifeCycle::Setup:
233  {
234  // Check for timeout. mTimeout == 0 means infinite timeout. NOTE: It
235  // feels like this code should go into the Test::loop() method (like
236  // the extra bit of code in TestOnce::loop()) because it seems like
237  // we could want the timeout to be configurable on a case by case
238  // basis. This would cause the testing() code to move down into a
239  // new again() virtual method dispatched from Test::loop(),
240  // analogous to once(). But let's keep the code here for now.
241  if (mTimeout > 0 && ellapsedSeconds() >= mTimeout) {
242  (*mCurrent)->expire();
243  } else {
244  (*mCurrent)->loop();
245 
246  // If test status is unresolved (i.e. still in LifeCycle::New
247  // state) after loop(), then this is a continuous testing() test
248  // case, so skip to the next test. Otherwise, stay on the current
249  // test so that the next iteration of runTest() can resolve the
250  // current test.
251  if ((*mCurrent)->getLifeCycle() == Test::LifeCycle::Setup) {
252  // skip to the next one, but keep current test in the list
253  mCurrent = (*mCurrent)->getNext();
254  }
255  }
256  }
257  break;
258  case Test::LifeCycle::Asserted:
259  switch ((*mCurrent)->getStatus()) {
260  case Test::Status::Skipped:
261  mSkippedCount++;
262  break;
263  case Test::Status::Passed:
264  mPassedCount++;
265  break;
266  case Test::Status::Failed:
267  mFailedCount++;
268  break;
269  case Test::Status::Expired:
270  mExpiredCount++;
271  break;
272  default:
273  // should never get here
274  mStatusErrorCount++;
275  break;
276  }
277  (*mCurrent)->teardown();
278  (*mCurrent)->setLifeCycle(Test::LifeCycle::Finished);
279  break;
280  case Test::LifeCycle::Finished:
281  (*mCurrent)->resolve();
282  // skip to the next one by taking current test out of the list
283  *mCurrent = *(*mCurrent)->getNext();
284  break;
285  }
286  }
