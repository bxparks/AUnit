54 void TestRunner::setLifeCycleMatchingPattern(const char* pattern,
55  Test::LifeCycle lifeCycle) {
56  // Do an implicit excludeAll() if the first filter is an include().
57  if (!hasBeenFiltered && lifeCycle == Test::LifeCycle::New) {
58  excludeAll();
59  }
60  hasBeenFiltered = true;
61 
62  size_t length = strlen(pattern);
63  if (length > 0 && pattern[length - 1] == '*') {
64  // prefix match
65  length--;
66  } else {
67  // exact match
68  length++;
69  }
70 
71  for (Test** p = Test::getRoot(); *p != nullptr; p = (*p)->getNext()) {
72  if ((*p)->getName().compareToN(pattern, length) == 0) {
73  (*p)->setLifeCycle(lifeCycle);
74  }
75  }
76 }

78 void TestRunner::setLifeCycleMatchingPattern(const char* testClass,
79  const char* pattern, Test::LifeCycle lifeCycle) {
80 
81  // Do an implicit excludeAll() if the first filter is an include().
82  if (!hasBeenFiltered && lifeCycle == Test::LifeCycle::New) {
83  excludeAll();
84  }
85  hasBeenFiltered = true;
86 
87  // The effective pattern is the join of testClass and pattern with a '_'
88  // delimiter. This must match the algorithm used by testF() and testingF().
89  // We use string_join() instead of String so that AUnit can avoid a direct
90  // dependency on the String class. AUnit thus avoids allocating any memory on
91  // the heap.
92  char fullPattern[kMaxPatternLength];
93  bool status = internal::string_join(fullPattern, kMaxPatternLength, '_',
94  testClass, pattern);
95  if (!status) return;
96 
97  setLifeCycleMatchingPattern(fullPattern, lifeCycle);
98 }

211 void TestRunner::processCommaList(
212  const char* const commaList, FilterType filterType) {
213 
214  const int kArgumentSize = 64;
215  const char* list = commaList;
216 
217  // Loop for each comma-separate list of words. Call the filtering command
218  // defined by the filterType.
219  char argument[kArgumentSize];
220  while (*list != '\0') {
221  const char* comma = strchr(list, ',');
222  int length = (comma) ? (comma - list) : strlen(list);
223 
224  // Copy each word of the comma-separated list.
225  if (length >= kArgumentSize - 1) {
226  length = kArgumentSize - 1;
227  }
228  memcpy(argument, list, length);
229  argument[length] = '\0';
230 
231  switch (filterType) {
232  case FilterType::kInclude:
233  include(argument);
234  break;
235  case FilterType::kExclude:
236  exclude(argument);
237  break;
238  case FilterType::kIncludeSub:
239  includesub(argument);
240  break;
241  case FilterType::kExcludeSub:
242  excludesub(argument);
243  break;
244  }
245 
246  list += (comma) ? length + 1 : length;
247  }
248 }

254 int TestRunner::parseFlags(int argc, const char* const* argv) {
255  int argc_original = argc;
256  shift(argc, argv);
257  while (argc > 0) {
258  if (argEquals(argv[0], "--include")) {
259  shift(argc, argv);
260  if (argc == 0) usageAndExit(1);
261  processCommaList(argv[0], FilterType::kInclude);
262  } else if (argEquals(argv[0], "--exclude")) {
263  shift(argc, argv);
264  if (argc == 0) usageAndExit(1);
265  processCommaList(argv[0], FilterType::kExclude);
266  } else if (argEquals(argv[0], "--includesub")) {
267  shift(argc, argv);
268  if (argc == 0) usageAndExit(1);
269  processCommaList(argv[0], FilterType::kIncludeSub);
270  } else if (argEquals(argv[0], "--excludesub")) {
271  shift(argc, argv);
272  if (argc == 0) usageAndExit(1);
273  processCommaList(argv[0], FilterType::kExcludeSub);
274  } else if (argEquals(argv[0], "--")) {
275  shift(argc, argv);
276  break;
277  } else if (argEquals(argv[0], "--help") || argEquals(argv[0], "-h")) {
278  usageAndExit(0);
279  break;
280  } else if (argv[0][0] == '-') {
281  fprintf(stderr, "Unknonwn flag '%s'\n", argv[0]);
282  usageAndExit(1);
283  } else {
284  break;
285  }
286  shift(argc, argv);
287  }
288 
289  return argc_original - argc;
290 }
