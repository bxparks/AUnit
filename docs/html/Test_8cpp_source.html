58 // Insert the current test case into the singly linked list, sorted by
59 // getName(). This is an O(N^2) algorithm, but should be good enough for
60 // small N ~= 100. If N becomes bigger than that, it's probably better to insert
61 // using an O(N) algorithm, then sort the elements later in TestRunner::run().
62 // Also, we don't increment a static counter here, because that would introduce
63 // another static initialization ordering problem.
64 void Test::insert() {
65  // Find the element p whose p->next sorts after the current test
66  Test** p = getRoot();
67  while (*p != nullptr) {
68  if (getName().compareTo((*p)->getName()) < 0) break;
69  p = &(*p)->mNext;
70  }
71  mNext = *p;
72  *p = this;
73 }

78  if (!isVerbosity(Verbosity::kTestAll)) return;
79 
80  Print* printer = Printer::getPrinter();
81  switch (mStatus)

83  case Test::Status::Passed:
84  if (isVerbosity(Verbosity::kTestPassed)) result = "\033[32m passed\033[37m.";
85  break;
86  case Test::Status::Failed:
87  if (isVerbosity(Verbosity::kTestFailed)) result = "\033[31m failed\033[37m.";
88  break;
89  case Test::Status::Skipped:
90  if (isVerbosity(Verbosity::kTestSkipped)) result = " skipped.";
91  break;
92  case Test::Status::Expired:
93  if (isVerbosity(Verbosity::kTestExpired)) result = "\033[33m failed\033[37m.";
94  break;
95  case Test::Status::Unknown:
96  break;
97  }
98 
99  if (result.length())
100  {
101  printer->print(F("Test"));
102  int spc = maxLength - mName.length() + 1;
103 
104  while(spc > 0)
105  {
106  printer->print(' ');
107  spc--;
108  }
109 
110  mName.print(printer);
111  printer->println(result.c_str());
112  }
113 }
